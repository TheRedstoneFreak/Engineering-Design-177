//Function to try to close the cabinet every few seconds
void mechanicalError() {
  int tryCounter = 0;
  while tryCounter < 3 {
    delay(mechanicalErrorDelay);
    if cabIsMoving() {
      continue;
    }
    cabClosed = closeCabinet();
    tryCounter++;
    if cabClosed {
      return false
    }
  }
  return true
}
